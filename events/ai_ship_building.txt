province_event = {
	id = 1111111
	title = "Shipbuilder builds Ironclad"
	desc = "Private investments has built a ironclad"
	
	trigger = {
		any_core = {
			is_coastal = yes
			port = yes
		}
		owner = {
			invention = ironclads
			NOT = { 
				OR = {
					invention = protected_cruisers
					invention = pre_dreadnought
				}
			}
			ai = yes
		}
	}
	
	mean_time_to_happen = {
		days = 20
	}
	
	option = {
		name = "Invest!"
		sub_unit = { 
			type = steam_transport
			value = 1
		}
		sub_unit = { 
			type = ironclad
			value = 1
		}
	}
}
